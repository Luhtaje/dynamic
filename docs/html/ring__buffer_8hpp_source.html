632 {
633 if (!std::is_nothrow_move_constructible<value_type>::value) return;
634
635 if (std::allocator_traits<allocator_type>::propagate_on_container_move_assignment::value)
636 {

763 iterator erase(const_iterator pos)
764 {
765 iterator returnIt(this, pos.getIndex());
766
767 if (pos != end() - 1)
768 {
769 // Move assign elements effectively overwriting the
770 for (auto it = returnIt; it < end() - 1; it++)
771 {
772 *it = std::move(*(it + 1));
773 }
774 ++returnIt;
775 }
776
777
778 // Last element is always destroyed.
779 m_allocator.destroy(&*(end() - 1));
780 decrement(m_headIndex);

782 return returnIt;

793 iterator erase(const_iterator first, const_iterator last)
794 {
795 iterator returnIt(this, first.getIndex());
796
797 const auto diff = std::distance(first, last);
798
799 // If the range does not reach the end, move elements.
800 if (last != end())
801 {
802 auto tempIt = returnIt;
803 // Move elements effectively overwriting the elements to be erased.
804 for (; tempIt + diff != end(); ++tempIt)
805 {
806 *tempIt = std::move(*(tempIt + diff));
807 }
808
809 for (; tempIt != end(); ++tempIt)
810 {
811 m_allocator.destroy(&*tempIt);
812 }
813 }
814 else
815 {
816 for (auto it = end() - diff; it != end(); it++)
817 {
818 m_allocator.destroy(&*(it));
819 }
820 }
821
822 decrement(m_headIndex, diff);
823 return returnIt;
824 }

1043 pointer data()
1044 {
1045 if(!size())
1046 {
1047 return m_data;
1048 }
1049
1050 // Data pointer for "do stuff and swap" idiom to provide strong excepion guarantee.
1051 auto tempData = m_allocator.allocate(m_capacity);
1052
1053 for (size_t i = 0; i < size(); i++)
1054 {
1055 m_allocator.construct(tempData + i, std::move(this->operator[](i)));
1056 m_allocator.destroy(&this->operator[](i));
1057 }
1058
1059 m_allocator.deallocate(m_data, size());
1060
1061 // If memory was allocated, the buffer matches beginning of physical memory.
1062 m_headIndex = size();
1063 m_tailIndex = 0;
1064
1065 // Assings the data from temp to original buffer. The resources from temp will be released when function goes out of scope.
1066 std::swap(tempData, m_data);
1067
1068 return m_data;
