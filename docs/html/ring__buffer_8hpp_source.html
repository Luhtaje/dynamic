618 ring_buffer(ring_buffer&& other) noexcept
619 {
620 if (!std::is_nothrow_move_constructible<value_type>::value) return;
621
622 if (std::allocator_traits<allocator_type>::propagate_on_container_move_assignment::value)
623 {

625 m_data = std::exchange(other.m_data, nullptr);

783 iterator erase(const_iterator pos)
784 {
785 iterator returnIt(this, pos.getIndex());
786
787 if (pos != end() - 1)
788 {
789 // Move assign elements effectively overwriting the
790 for (auto it = returnIt; it < end() - 1; it++)
791 {
792 *it = std::move(*(it + 1));
793 }
794 ++returnIt;
795 }

799 m_allocator.destroy(&*(end() - 1));
800 decrement(m_headIndex);

813 iterator erase(const_iterator first, const_iterator last)
814 {
815 iterator returnIt(this, first.getIndex());
816
817 const auto diff = std::distance(first, last);
818
819 // If the range does not reach the end, move elements.
820 if (last != end())
821 {
822 auto tempIt = returnIt;
823 // Move elements effectively overwriting the elements to be erased.
824 for (; tempIt + diff != end(); ++tempIt)
825 {
826 *tempIt = std::move(*(tempIt + diff));
827 }
828
829 for (; tempIt != end(); ++tempIt)
830 {
831 m_allocator.destroy(&*tempIt);
832 }
833 }
834 else
835 {
836 for (auto it = end() - diff; it != end(); it++)
837 {
838 m_allocator.destroy(&*(it));
839 }
840 }
841
842 decrement(m_headIndex, diff);
843 return returnIt;

1063 pointer data()
1064 {
1065 if(!size())
1066 {
1067 return m_data;
1068 }
1069
1070 // Data pointer for "do stuff and swap" idiom to provide strong excepion guarantee.
1071 auto tempData = m_allocator.allocate(m_capacity);
1072
1073 for (size_t i = 0; i < size(); i++)
1074 {
1075 m_allocator.construct(tempData + i, std::move(this->operator[](i)));
1076 m_allocator.destroy(&this->operator[](i));
1077 }
1078
1079 m_allocator.deallocate(m_data, size());
1080
1081 // If memory was allocated, the buffer matches beginning of physical memory.
1082 m_headIndex = size();
1083 m_tailIndex = 0;
1084
1085 // Assings the data from temp to original buffer. The resources from temp will be released when function goes out of scope.
1086 std::swap(tempData, m_data);
1087
1088 return m_data;
1089 }

1447 template<typename InsertValue>
1448 iterator insertBase(const_iterator pos, const size_type amount, InsertValue&& value)
1449 {
1450 iterator it(this, pos.getIndex());
1451
1452 if (!amount) return it;
1453
1454 validateCapacity(amount);
1455
1456 for (size_type i = 0; i < amount; i++)
1457 {
1458 // Construct element at the end.
1459 m_allocator.construct(&*end(), std::forward<InsertValue>(value));
1460 increment(m_headIndex);
1461 }
1462
1463 // Rotate elements from the back into pos.
1464 std::rotate(it, end() - amount, end());
1465
1466 return it;
1467 }

1489 increment(m_headIndex);

1492 iterator it(this, pos.getIndex());
1493 std::rotate(it, end() - amount, end());
1494
